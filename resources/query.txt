Find total of each student.
select sum(marks), std_id, (select name from Students where id = std_id) from Scores group by std_id;

select 
    sum(marks) as Total, 
    std_id STD_ID, 
    (select name from Students where id = std_id) as Name 
from Scores group by std_id;


Name students who scored more than average (72.7143).
select avg(marks) from Scores; -- 72.7143


select 
    avg(marks) as Total, 
    std_id STD_ID, 
    (select name from Students where id = std_id) as Name 
from Scores group by std_id having avg(marks) > (select avg(marks) from Scores);


Find highest scorer.
select max(score) from (select sum(marks) as score from Scores group by std_id) as t;

desc ->

Find second highest scorer.
select 
    sum(marks) as score, 
    (select name from Students where id = std_id) as Name 
from Scores group by std_id order by score desc limit 1 offset 1;
